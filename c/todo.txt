* use: https://github.com/petewarden/dstkdata
  mlr --from ../data/big.dkvp put -q '
    #weight = $y;
    #weight = $y * $y;
    weight = $i;

    @sumwx[$a] += weight * $i;
    @sumw[$a] += weight;

    @sumx[$a] += $i;
    @sumn[$a] += 1;

    end {
      map wmean = {};
      map mean  = {};
      for (a in @sumwx) {
        wmean[a] = @sumwx[a] / @sumw[a]
      }
      for (a in @sumx) {
        mean[a] = @sumx[a] / @sumn[a]
      }
      emit wmean, "a";
      emit mean, "a";
    }'
